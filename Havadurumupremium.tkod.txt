        Don "⛅";
    EgerBitti
    
    Eger MetinIceriyorMu(kucukDurum, "bulut") Veya MetinIceriyorMu(kucukDurum, "cloud") Ise
        Don "☁️";
    EgerBitti
    
    Eger MetinIceriyorMu(kucukDurum, "yağmur") Veya MetinIceriyorMu(kucukDurum, "rain") Ise
        Don "🌧️";

    Eger MetinIceriyorMu(kucukDurum, "kar") Veya MetinIceriyorMu(kucukDurum, "snow") Ise
        Don "❄️";
    EgerBitti
    
    Eger MetinIceriyorMu(kucukDurum, "fırtına") Veya MetinIceriyorMu(kucukDurum, "storm") Ise
        Don "⛈️";
    EgerBitti

// ========== API'DEN VERİ ÇEKME ==========
Fonksiyon havaDurumuGetir(sehirAdi)
    Eger yukleniyor == Dogru Ise
        Don;
    EgerBitti
    
    yukleniyor = Dogru;
    sehir = sehirAdi;
    panelCiz();
    
    Metin url = "https://wttr.in/" + sehirAdi + "?format=j1";
    
    Yaz("📍 Hava durumu alınıyor: " + sehirAdi);
    
    Metin cevap = VeriCek(url);
    
    Eger cevap == "" Ise
        durum = "Veri alınamadı";
        ikon = "❌";
        sicaklik = 0;
        hissedilen = 0;
        nem = 0;
        ruzgar = 0;
        sesCal("alarm");
    Degilse
        Metin sicaklikStr = JSONOku(cevap, "current_condition");
        Eger sicaklikStr == "" Ise
            durum = "Şehir bulunamadı";
            ikon = "❓";
            sicaklik = 0;
        Degilse
            Metin temp = cevap;
            
            // Sıcaklık
            Sayi tempIndex = MetinIndeks(temp, "\"temp_C\"");
            Eger tempIndex >= 0 Ise
                Metin tempStr = MetinKes(temp, tempIndex + 9, tempIndex + 14);
                tempStr = MetinDegistir(tempStr, "\"", "");
                tempStr = MetinDegistir(tempStr, ",", "");
                tempStr = MetinDegistir(tempStr, ":", "");
                tempStr = MetinBosYer(tempStr);
                sicaklik = SayiyaKonvert(tempStr);
            EgerBitti
            
            // Hissedilen
            Sayi feelsIndex = MetinIndeks(temp, "\"FeelsLikeC\"");
            Eger feelsIndex >= 0 Ise
                Metin feelsStr = MetinKes(temp, feelsIndex + 12, feelsIndex + 17);
                feelsStr = MetinDegistir(feelsStr, "\"", "");
                feelsStr = MetinDegistir(feelsStr, ",", "");
                feelsStr = MetinDegistir(feelsStr, ":", "");
                feelsStr = MetinBosYer(feelsStr);
                hissedilen = SayiyaKonvert(feelsStr);
            EgerBitti
            
            // Nem
            Sayi nemIndex = MetinIndeks(temp, "\"humidity\"");
            Eger nemIndex >= 0 Ise
                Metin nemStr = MetinKes(temp, nemIndex + 10, nemIndex + 15);
                nemStr = MetinDegistir(nemStr, "\"", "");
                nemStr = MetinDegistir(nemStr, ",", "");
                nemStr = MetinDegistir(nemStr, ":", "");
                nemStr = MetinBosYer(nemStr);
                nem = SayiyaKonvert(nemStr);
            EgerBitti
            
            // Rüzgar
            Sayi ruzgarIndex = MetinIndeks(temp, "\"windspeedKmph\"");
            Eger ruzgarIndex >= 0 Ise
                Metin ruzgarStr = MetinKes(temp, ruzgarIndex + 15, ruzgarIndex + 20);
                ruzgarStr = MetinDegistir(ruzgarStr, "\"", "");
                ruzgarStr = MetinDegistir(ruzgarStr, ",", "");
                ruzgarStr = MetinDegistir(ruzgarStr, ":", "");
                ruzgarStr = MetinBosYer(ruzgarStr);
                ruzgar = SayiyaKonvert(ruzgarStr);
            EgerBitti
            
            // Durum
            Sayi durumIndex = MetinIndeks(temp, "\"weatherDesc\"");
            Eger durumIndex >= 0 Ise
                Sayi valueIndex = MetinIndeks(temp, "\"value\"", durumIndex);
                Eger valueIndex >= 0 Ise
                    Metin durumStr = MetinKes(temp, valueIndex + 8, valueIndex + 40);
                    durumStr = MetinDegistir(durumStr, "\"", "");
                    durumStr = MetinDegistir(durumStr, ",", "");
                    durumStr = MetinBosYer(durumStr);
                    Eger durumStr != "" Ise
                        durum = durumStr;
                    EgerBitti
                EgerBitti
            EgerBitti
            
            ikon = ikonBelirle(durum);
            sesCal("ding");
        EgerBitti
    EgerBitti
    
    yukleniyor = Yanlis;
    panelCiz();
    populerPanelCiz();
FonksiyonBitti

// ========== POPÜLER ŞEHİR PANELİ ==========
Fonksiyon populerPanelCiz()
    Eger geceModu == Dogru Ise
        KanvasArkaplan(populerPanel, "#0d1117");
    Degilse
        KanvasArkaplan(populerPanel, "#1a2a3a");
    EgerBitti
    
    Sayi i = 0;
    Sayi basX = 10;
    Sayi basY = 10;
    
    Tekrarla 5 Kez
        Eger i < DiziUzunluk(populerSehirler) Ise
            Sayi bx = basX + (i * 65);
            
            Eger i == seciliPopuler Ise
                KanvasDikdortgen(populerPanel, bx - 2, basY - 2, 60, 32, "#1f6feb");
            EgerBitti
            
            KanvasDikdortgen(populerPanel, bx, basY, 56, 28, "#2d3748");
            KanvasMetin(populerPanel, populerSehirler[i], bx + 5, basY + 19, "#ffffff", "bold 9px Segoe UI");
        EgerBitti
        i = i + 1;
    TekrarlaBitti
    
    i = 5;
    basY = 42;
    Tekrarla 5 Kez
        Eger i < DiziUzunluk(populerSehirler) Ise
            Sayi bx = basX + ((i - 5) * 65);
            
            Eger i == seciliPopuler Ise
                KanvasDikdortgen(populerPanel, bx - 2, basY - 2, 60, 32, "#1f6feb");
            EgerBitti
            
            KanvasDikdortgen(populerPanel, bx, basY, 56, 28, "#2d3748");
            KanvasMetin(populerPanel, populerSehirler[i], bx + 5, basY + 19, "#ffffff", "bold 9px Segoe UI");
        EgerBitti
        i = i + 1;
    TekrarlaBitti
FonksiyonBitti

// ========== POPÜLER ŞEHİR TIKLAMA ==========
Fonksiyon populerTiklandi(mx, my)
    Sayi i = 0;
    Sayi basX = 30;
    Sayi basY = 230;
    
    Tekrarla 5 Kez
        Eger mx >= basX + (i * 65) Ve mx <= basX + (i * 65) + 56 Ise
            Eger my >= basY + 10 Ve my <= basY + 38 Ise
                Eger i < DiziUzunluk(populerSehirler) Ise
                    seciliPopuler = i;
                    gsSehir.deger = populerSehirler[i];
                    havaDurumuGetir(populerSehirler[i]);
                EgerBitti
            EgerBitti
        EgerBitti
        i = i + 1;
    TekrarlaBitti
    
    i = 5;
    Tekrarla 5 Kez
        Eger mx >= basX + ((i - 5) * 65) Ve mx <= basX + ((i - 5) * 65) + 56 Ise
            Eger my >= basY + 32 Ve my <= basY + 60 Ise
                Eger i < DiziUzunluk(populerSehirler) Ise
                    seciliPopuler = i;
                    gsSehir.deger = populerSehirler[i];
                    havaDurumuGetir(populerSehirler[i]);
                EgerBitti
            EgerBitti
        EgerBitti
        i = i + 1;
    TekrarlaBitti
FonksiyonBitti
